Upload Component Implementation

// components/media/MediaUploader.tsx import React, { useState, useCallback } from 'react'; import { useDropzone } from 'react-dropzone'; interface MediaUploaderProps { onUpload: (files: File[]) => void; maxFiles?: number; maxSize?: number; acceptedTypes?: string[]; } const MediaUploader: React.FC<MediaUploaderProps> = ({ onUpload, maxFiles = 1, maxSize = 10485760, // 10MB acceptedTypes = ['image/jpeg', 'image/png', 'image/webp'] }) => { const [uploading, setUploading] = useState(false); const [progress, setProgress] = useState(0); const onDrop = useCallback(async (acceptedFiles: File[]) => { setUploading(true); try { const uploadPromises = acceptedFiles.map(file => uploadFileWithProgress(file, setProgress) ); const uploadResults = await Promise.all(uploadPromises); onUpload(uploadResults); } catch (error) { console.error('Upload failed:', error); } finally { setUploading(false); setProgress(0); } }, [onUpload]); const { getRootProps, getInputProps, isDragActive } = useDropzone({ onDrop, maxFiles, maxSize, accept: acceptedTypes.reduce((acc, type) => ({ ...acc, [type]: [] }), {}) }); return ( <div {...getRootProps()} className={`upload-container ${isDragActive ? 'dragover' : ''}`} > <input {...getInputProps()} /> {uploading ? ( <UploadProgress progress={progress} /> ) : ( <UploadPrompt isDragActive={isDragActive} /> )} </div> ); };

Image Processing Pipeline

// utils/imageProcessing.ts export const processImage = async (file: File) => { const canvas = document.createElement('canvas'); const ctx = canvas.getContext('2d')!; const img = new Image(); return new Promise((resolve) => { img.onload = () => { // Smart cropping to 16:9 aspect ratio const targetRatio = 16 / 9; const sourceRatio = img.width / img.height; let cropWidth, cropHeight, cropX, cropY; if (sourceRatio > targetRatio) { cropHeight = img.height; cropWidth = cropHeight * targetRatio; cropX = (img.width - cropWidth) / 2; cropY = 0; } else { cropWidth = img.width; cropHeight = cropWidth / targetRatio; cropX = 0; cropY = (img.height - cropHeight) / 2; } canvas.width = 1200; canvas.height = 630; ctx.drawImage( img, cropX, cropY, cropWidth, cropHeight, 0, 0, 1200, 630 ); canvas.toBlob(resolve, 'image/webp', 0.85); }; img.src = URL.createObjectURL(file); }); };

Media API Integration

// API Routes for Media Management // pages/api/media/upload.ts export default async function handler(req: NextApiRequest, res: NextApiResponse) { if (req.method !== 'POST') { return res.status(405).json({ message: 'Method not allowed' }); } try { const form = new IncomingForm(); const [fields, files] = await form.parse(req); const file = files.file as File; const optimizedFile = await optimizeImage(file); // Upload to CDN const cdnUrl = await uploadToCDN(optimizedFile); // Generate alt text using AI const altText = await generateAltText(optimizedFile); // Save to database const mediaRecord = await prisma.media.create({ data: { filename: file.name, url: cdnUrl, altText: altText, size: optimizedFile.size, mimeType: optimizedFile.type, userId: req.user.id } }); res.status(200).json({ media: mediaRecord }); } catch (error) { res.status(500).json({ error: 'Upload failed' }); } }
