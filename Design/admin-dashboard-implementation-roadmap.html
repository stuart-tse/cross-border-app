API Endpoints

User Management APIs

GET /api/admin/users - List all users with filters
PUT /api/admin/users/[id] - Update user details
DELETE /api/admin/users/[id] - Deactivate user
POST /api/admin/users/bulk - Bulk operations

Verification APIs

GET /api/admin/verification/pending - Pending verifications
PUT /api/admin/verification/[id]/approve - Approve verification
PUT /api/admin/verification/[id]/reject - Reject verification

Analytics APIs

GET /api/admin/analytics/metrics - Dashboard metrics
GET /api/admin/analytics/charts - Chart data
GET /api/admin/analytics/reports - Generate reports

State Management Pattern

// Admin Dashboard Context
const AdminContext = createContext();

const AdminProvider = ({ children }) => {
  const [state, dispatch] = useReducer(adminReducer, {
    users: { clients: [], drivers: [], editors: [] },
    metrics: { totalUsers: 0, revenue: 0, trips: 0 },
    filters: { search: '', status: 'all', dateRange: null },
    ui: { loading: false, selectedUsers: [], activeModal: null }
  });

  return (
    <AdminContext.Provider value={{ state, dispatch }}>
      {children}
    </AdminContext.Provider>
  );
};

// Usage in components
const { state, dispatch } = useContext(AdminContext);
